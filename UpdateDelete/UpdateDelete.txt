	-- Operadores
DECLARE @A INT = 10;
SET @A += 5;   -- O mesmo que SET @A = @A + 5;
PRINT @A;
SET @A -= 2;   -- O mesmo que SET @A = @A - 2;
PRINT @A;
SET @A *= 4;   -- O mesmo que SET @A = @A * 4;
PRINT @A;
SET @A /= 2;   -- O mesmo que SET @A = @A / 2;
PRINT @A;
GO

-- OU
UPDATE EMPREGADOS

-- codigo 5

SET NUM_DEPEND = NUM_DEPEND + 2
WHERE CODFUN = 5;
-- OU
UPDATE EMPREGADOS
SET NUM_DEPEND += 2
WHERE CODFUN = 5;
